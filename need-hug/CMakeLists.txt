cmake_minimum_required(VERSION 3.7)

# Root project settings
get_filename_component(FOLDER_NAME ${CMAKE_CURRENT_SOURCE_DIR} NAME)
set(MODULE_NAME ${FOLDER_NAME})
project(${MODULE_NAME})

# Include
set(EXTERNAL_INCLUDE_DIRECTORIES "")

# Libraries
set(INTERNAL_LIBRARIES need-hug-lib)
set(EXTERNAL_LIBRARIES "")

# Set compiler options for target
set(DEBUG_OPTIONS "")
set(RELEASE_OPTIONS "")
set(COMPILE_OPTIONS $<$<CONFIG:DEBUG>:${DEBUG_OPTIONS}> $<$<CONFIG:RELEASE>:${RELEASE_OPTIONS}>)
	
# message("EXTERNAL_INCLUDE_DIRECTORIES ${EXTERNAL_INCLUDE_DIRECTORIES}")
# message("MODULE_NAME ${MODULE_NAME}")
# message("CMAKE_CURRENT_SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR}")
# message("COMPILE_OPTIONS ${COMPILE_OPTIONS}")
# message("INTERNAL_LIBRARIES ${INTERNAL_LIBRARIES}")
# message("EXTERNAL_LIBRARIES ${EXTERNAL_LIBRARIES}")

mz_add_executable("${MODULE_NAME}" "${CMAKE_CURRENT_SOURCE_DIR}" "${COMPILE_OPTIONS}" "${INTERNAL_LIBRARIES}" "${EXTERNAL_LIBRARIES}" "${EXTERNAL_INCLUDE_DIRECTORIES}")
if(WIN32)
	set_property(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT ${FOLDER_NAME})
endif(WIN32)
